# Page Wrapper

Every page in Toast will be wrapped with a component located at `src/page-wrapper.js`. This can be used to add context providers, CSS, and other global components.

```jsx
/** @jsx h */
import { h } from "preact";
import { Helmet } from "react-helmet";
import { MDXProvider } from "@mdx-js/preact";

export default ({ children, ...props }) => {
  const components = {
    p: props => (...) // Customize default mdx components
    Callout, // Add custom components as global shortcodes
  }

  return (
    <div>
      <Helmet>{/* CSS, SEO Metadata, etc */}</Helmet>
      <header>
        <h1>New Toast Site</h1>
      </header>
      <MDXProvider components={components}>{children}</MDXProvider>
    </div>
  );
};
```
